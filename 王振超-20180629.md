#### LeetCode

[Two Sum](https://leetcode-cn.com/problems/two-sum/description/)

```go
func twoSum(nums []int, target int) []int {
	var length = len(nums)
	for i := 0; i < length; i++ {
		for j := 0; j < length; j++ {
			if nums[i] + nums[j] == target && i != j {
				return []int{i, j}
			}
		}
	}
	return []int{}
}
```
